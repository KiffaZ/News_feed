  <script>

    const feeds = [

      { name: "Лента.ру", url: "https://lenta.ru/rss/news" },

      { name: "РИА Новости", url: "https://ria.ru/export/rss2/all/index.xml" },

      { name: "РБК", url: "https://rssexport.rbc.ru/rbcnews/news/30/full.rss" },

      { name: "ТАСС", url: "https://tass.ru/rss/v2.xml" },

      { name: "Коммерсантъ", url: "https://www.kommersant.ru/RSS/news.xml" },

      { name: "Газета.Ru", url: "https://www.gazeta.ru/export/rss2/all/index.xml" }

    ];


    const CORS_PROXY = "https://api.allorigins.win/get?url=";


    // Безопасная работа с localStorage

    function safeLocalStorageGet(key, defaultValue = null) {

      try {

        const value = localStorage.getItem(key);

        return value !== null ? value : defaultValue;

      } catch (e) {

        return defaultValue;

      }

    }


    function safeLocalStorageSet(key, value) {

      try {

        localStorage.setItem(key, value);

      } catch (e) {

        // Игнорируем ошибку (например, при file://)

      }

    }


    function toggleTheme() {

      document.documentElement.classList.toggle("dark");

      const isDark = document.documentElement.classList.contains("dark");

      document.getElementById("theme-icon").className = isDark 

        ? "fa-solid fa-sun text-xl" 

        : "fa-solid fa-moon text-xl";

      safeLocalStorageSet("theme", isDark ? "dark" : "light");

    }


    async function loadNews() {

      const container = document.getElementById("news-container");

      const loading = document.getElementById("loading");

      container.innerHTML = "";

      loading.style.display = "block";


      let allNews = [];

      const sourcesShown = new Set();


      for (const feed of feeds) {

        try {

          const response = await fetch(CORS_PROXY + encodeURIComponent(feed.url));

          const data = await response.json();

          const parser = new DOMParser();

          const doc = parser.parseFromString(data.contents, "text/xml");

          

          const items = doc.querySelectorAll("item");

          

          items.forEach(item => {

            const title = item.querySelector("title")?.textContent || "Без заголовка";

            const link = item.querySelector("link")?.textContent || "#";

            let description = (item.querySelector("description")?.textContent || "").replace(/<[^>]+>/g, "").slice(0, 160);

            if (description) description += "...";

            

            const pubDate = item.querySelector("pubDate")?.textContent || "";

            const date = pubDate ? new Date(pubDate) : new Date();


            allNews.push({ title, link, description, date, source: feed.name, timestamp: date.getTime() });

            sourcesShown.add(feed.name);

          });

        } catch(e) {

          console.warn("Не удалось загрузить:", feed.name);

        }

      }


      allNews.sort((a, b) => b.timestamp - a.timestamp);


      // Убираем дубликаты

      const seen = new Set();

      const uniqueNews = allNews.filter(news => {

        if (seen.has(news.title)) return false;

        seen.add(news.title);

        return true;

      });


      loading.style.display = "none";


      uniqueNews.slice(0, 40).forEach((news, index) => {

        const card = document.createElement("div");

        card.className = "news-card bg-white dark:bg-zinc-900 rounded-3xl p-6 border border-gray-200 dark:border-zinc-800";

        card.innerHTML = `

          <div class="flex justify-between text-xs mb-3">

            <span class="px-3 py-1 bg-purple-100 dark:bg-purple-900 text-purple-700 dark:text-purple-300 rounded-full">${news.source}</span>

            <span class="text-gray-500">${news.date.toLocaleDateString('ru-RU')} ${news.date.toLocaleTimeString('ru-RU', {hour:'2-digit', minute:'2-digit'})}</span>

          </div>

          <a href="${news.link}" target="_blank" class="block hover:text-purple-600 dark:hover:text-purple-400 transition-colors">

            <h2 class="text-xl font-semibold leading-tight mb-3">${news.title}</h2>

            <p class="text-gray-600 dark:text-gray-400">${news.description || ''}</p>

          </a>

        `;

        container.appendChild(card);


        if ((index + 1) % 5 === 0) {

          const ad = document.createElement("div");

          ad.className = "ad-placeholder rounded-3xl p-6 my-8 text-center";

          ad.innerHTML = `<p class="text-purple-600 dark:text-purple-400 text-sm font-medium">Место под блок Яндекс.РСЯ</p>`;

          container.appendChild(ad);

        }

      });


      document.getElementById("sources-list").innerHTML = Array.from(sourcesShown)

        .map(s => `<div class="flex items-center gap-2"><i class="fa-solid fa-check text-green-500"></i> ${s}</div>`).join("");

    }


    function refreshNews() {

      loadNews();

    }


    // Инициализация

    window.onload = () => {

      // Восстанавливаем тему безопасно

      if (safeLocalStorageGet("theme") === "light") {

        document.documentElement.classList.remove("dark");

        document.getElementById("theme-icon").className = "fa-solid fa-moon text-xl";

      } else {

        document.getElementById("theme-icon").className = "fa-solid fa-sun text-xl";

      }


      loadNews();


      // Автообновление каждые 10 минут

      setInterval(loadNews, 600000);

    };
